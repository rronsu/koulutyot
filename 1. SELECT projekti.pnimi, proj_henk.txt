1. SELECT projekti.pnimi, proj_henk.tunnit FROM projekti JOIN proj_henk ON projekti.ptun = proj_henk.ptun WHERE proj_henk.htun = 2245;
2. SELECT projekti.pnimi, proj_henk.htun, proj_henk.tunnit, proj_henk.tunnit_suun FROM projekti JOIN proj_henk ON projekti.ptun = proj_henk.ptun ORDER BY projekti.pnimi;
3. SELECT osasto.ostun, osasto.osnimi, henkilo.snimi, henkilo.enimi, henkilo.kunta, henkilo.palkka FROM osasto JOIN henkilo ON osasto.ostun = henkilo.ostun WHERE henkilo.palkka = 2800 ORDER BY osasto.osnimi, henkilo.snimi, henkilo.enimi;
4.SELECT projekti.pnimi, henkilo.enimi, henkilo.snimi, proj_henk.tunnit FROM projekti JOIN proj_henk ON projekti.ptun = proj_henk.ptun JOIN henkilo ON proj_henk.htun = henkilo.htun ORDER BY projekti.pnimi, henkilo.enimi, henkilo.snimi;
5.SELECT projekti.ptun AS ptun, projekti.pnimi, COALESCE(SUM(proj_henk.tunnit), 0) AS yht FROM projekti LEFT JOIN proj_henk ON projekti.ptun = proj_henk.ptun GROUP BY projekti.ptun, projekti.pnimi ORDER BY projekti.ptun;
6.SELECT projekti.ptun AS ptun, projekti.pnimi, SUM(proj_henk.tunnit) AS tunnit, SUM(proj_henk.tunnit - proj_henk.tunnit_suun) AS erotus FROM projekti JOIN proj_henk ON projekti.ptun = proj_henk.ptun JOIN henkilo ON proj_henk.htun = henkilo.htun WHERE henkilo.kunta IN ('HELSINKI', 'KUOPIO') GROUP BY projekti.ptun, projekti.pnimi ORDER BY projekti.ptun;
7.SELECT osasto.ostun AS tunnus, osasto.osnimi, MIN(henkilo.palkka) AS min, MAX(henkilo.palkka) AS max, AVG(henkilo.palkka) AS ka FROM osasto LEFT JOIN henkilo ON osasto.ostun = henkilo.ostun GROUP BY osasto.ostun, osasto.osnimi ORDER BY osasto.ostun;
